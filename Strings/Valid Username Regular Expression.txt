import java.util.Scanner;
class UsernameValidator {
    /*
     * Write regular expression here.
     */
     
     /*
     * ^ ==> Start of the string (ensures we check from the first character).
     * [A-Za-z] ==> First character must be an alphabet letter (uppercase or lowercase).
     * [A-Za-z0-9_] ==> After the first character can be an alphabet letter (upper or lower), any number between 
     * 0-9 and also the underscore sign (_).
     * {7,29} ==> The username consists of less than 8 or greater than 30 characters.
     * $ ==>     End of the string (ensures no extra characters).
     */
    public static final String regularExpression = "^[A-Za-z][A-Za-z0-9_]{7,29}$";
}


public class Solution {
    private static final Scanner scan = new Scanner(System.in);
    
    public static void main(String[] args) {
        int n = Integer.parseInt(scan.nextLine());
        while (n-- != 0) {
            String userName = scan.nextLine();

            if (userName.matches(UsernameValidator.regularExpression)) {
                System.out.println("Valid");
            } else {
                System.out.println("Invalid");
            }           
        }
    }
}
