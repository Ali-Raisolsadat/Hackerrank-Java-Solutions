import java.util.*;

public class Solution {

    public static boolean canWin(int leap, int[] game) {
        // Return true if you can win the game; otherwise, return false.
        return canWinFrom(0, leap, game, new boolean[game.length]);
    }
    
    private static boolean canWinFrom(int i, int leap, int[] game, boolean[] visited) {
        // base case: out of bounds or on a 1 or already visted
        if (i < 0 || game[i] == 1 || visited[i]) {
            return false;
        }
        
        // win condition
        if (i >= game.length-1 || i + leap >= game.length) {
            return true;
        }
        
        visited[i] = true;
        
        // try all valid moves: step forward, leap or back
        return canWinFrom(i + 1, leap, game, visited) ||
                canWinFrom(i + leap, leap, game, visited) ||
                canWinFrom(i - 1, leap, game, visited);
    }

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int q = scan.nextInt();
        while (q-- > 0) {
            int n = scan.nextInt();
            int leap = scan.nextInt();
            
            int[] game = new int[n];
            for (int i = 0; i < n; i++) {
                game[i] = scan.nextInt();
            }

            System.out.println( (canWin(leap, game)) ? "YES" : "NO" );
        }
        scan.close();
    }
}
